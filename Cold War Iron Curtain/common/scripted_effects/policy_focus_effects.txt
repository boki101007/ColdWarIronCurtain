generate_tax_info = {
    set_variable = {
        laffer_weighted_average_tax_rate = modifier@tax_rate_lower_class
    }
    add_to_variable = {
        laffer_weighted_average_tax_rate = modifier@tax_rate_upper_class
    }
    add_to_variable = {
        laffer_weighted_average_tax_rate = modifier@tax_rate_corporate
    }
    add_to_variable = {
        laffer_weighted_average_tax_rate = modifier@tax_rate_middle_class
    }
    divide_variable = {
        laffer_weighted_average_tax_rate = 4
    }

    #Run Laffer Curve Here
    laffer_curve_function = yes
    add_to_variable = {
        effective_tax_rate = modifier@modifier_tax_efficiacy 
    }

    #Get Tax Per Rate and then Efficiency Scores

    #multiply_variable = {
    #    tax_display = effective_tax_rate
    #}
    #multiply_variable = {
    #    tax_base = effective_tax_rate
    #}

    set_variable = {
        actual_tax_rate_adjusted = actual_tax_rate
    }
    multiply_variable = {
        actual_tax_rate_adjusted = effective_tax_rate
    }

    set_variable = {
        effective_tax_rate_display = actual_tax_rate
    }
    multiply_variable = {
        effective_tax_rate_display = 100
    }
    multiply_variable = {
        effective_tax_rate_display = 2
    }
    round_variable = effective_tax_rate_display
    add_to_variable = {
        effective_tax_rate_display = 89
    }
    Laffer_Graph_Financial_Tab_Update = yes
}
